cd /home/develop/googletest/
mkdir build; cd build
cmake -DCMAKE_INSTALL_PREFIX=/home/develop/local/ ..
make  && make install

新建文件TestRedisGtest.cpp：
#include "gtest/gtest.h"

int Add(int a, int b)
{
    return a + b;
}

TEST(AddTest, MyAddTest)
{
    EXPECT_EQ(2, Add(1, 1));
    EXPECT_EQ(6, Add(2, 4));
}

int main(int argc, char** argv)
{
    testing::InitGoogleTest(&argc, argv);
    return RUN_ALL_TESTS();
}


编译程序:
xlC_r -q64 -g -I/home/develop/local/include -c TestRedisGtest.cpp -o TestRedisGtest.o
xlC_r -o test_gtest TestRedisGtest.o -L/home/develop/local/lib -lgtest
